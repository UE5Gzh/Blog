---
import BaseLayout from './BaseLayout.astro';
import '../styles/post.css';
import TableOfContents from '../components/TableOfContents.astro';
import PostNavigation from '../components/PostNavigation.astro';
import { addCopyButtons } from '../components/CopyButton.astro';
import type { CollectionEntry } from 'astro:content';

interface Props {
  title: string;
  date: Date;
  description: string;
  tags: string[];
  headings?: { depth: number; slug: string; text: string }[];
  minutesRead?: string;
  posts?: CollectionEntry<'posts'>[];
  currentSlug?: string;
}

const {
  title,
  date,
  description,
  tags,
  headings = [],
  minutesRead,
  posts = [],
  currentSlug
} = Astro.props;

const formattedDate = new Date(date).toLocaleDateString('en-US', {
  year: 'numeric',
  month: 'long',
  day: 'numeric'
});

// Filter headings for TOC (H2 and H3 only)
const tocHeadings = headings.filter(h => h.depth === 2 || h.depth === 3);
---

<BaseLayout title={title} description={description}>
  <article class="post">
    <a href="/" class="back-link">
      <svg xmlns="http://www.w3.org/2000/svg" width="16" height="16" viewBox="0 0 24 24" fill="none" stroke="currentColor" stroke-width="2" stroke-linecap="round" stroke-linejoin="round">
        <polyline points="15 18 9 12 15 6"></polyline>
      </svg>
      Back to posts
    </a>

    <header>
      <h1>{title}</h1>
      <div class="post-meta">
        <time datetime={date.toISOString()}>
          <svg xmlns="http://www.w3.org/2000/svg" width="14" height="14" viewBox="0 0 24 24" fill="none" stroke="currentColor" stroke-width="2" stroke-linecap="round" stroke-linejoin="round" style="margin-right: 4px;">
            <rect x="3" y="4" width="18" height="18" rx="2" ry="2"></rect>
            <line x1="16" y1="2" x2="16" y2="6"></line>
            <line x1="8" y1="2" x2="8" y2="6"></line>
            <line x1="3" y1="10" x2="21" y2="10"></line>
          </svg>
          {formattedDate}
        </time>
        {minutesRead && (
          <span class="reading-time">
            <svg xmlns="http://www.w3.org/2000/svg" width="14" height="14" viewBox="0 0 24 24" fill="none" stroke="currentColor" stroke-width="2" stroke-linecap="round" stroke-linejoin="round" style="margin-right: 4px;">
              <circle cx="12" cy="12" r="10"></circle>
              <polyline points="12 6 12 12 16 14"></polyline>
            </svg>
            {minutesRead}
          </span>
        )}
        {tags && tags.length > 0 && (
          <div class="tags">
            {tags.map(tag => (
              <span class="tag">{tag}</span>
            ))}
          </div>
        )}
      </div>

      {/* Table of Contents - only show if 3+ headings */}
      {tocHeadings.length >= 3 && <TableOfContents headings={tocHeadings} />}
    </header>

    <div class="post-content">
      <slot />
    </div>

    {/* Previous/Next Navigation */}
    {posts.length > 0 && currentSlug && (
      <PostNavigation posts={posts} currentSlug={currentSlug} />
    )}
  </article>
</BaseLayout>

{/* Copy button initialization - runs on every page load including view transitions */}
<script>

  }

```astro
---
import { getCollection } from 'astro:content';
import PostLayout from '../../layouts/PostLayout.astro';

// Generate static paths for all posts
export async function getStaticPaths() {
  const posts = await getCollection('posts');
  return posts.map(post => ({
    params: { slug: post.slug },
    props: { post },
  }));
}

// Get the post entry
const { post } = Astro.props;

// Render to get headings and reading time
const { Content, headings, remarkPluginFrontmatter } = await post.render();

// Get all posts for navigation
const allPosts = await getCollection('posts');

// Extract reading time from remark plugin frontmatter
const minutesRead = remarkPluginFrontmatter?.minutesRead;
---

<PostLayout
  title={post.data.title}
  date={post.data.date}
  description={post.data.description}
  tags={post.data.tags}
  headings={headings}
  minutesRead={minutesRead}
  posts={allPosts}
  currentSlug={post.slug}
>
  <Content />
</PostLayout>
```

The key change is calling `post.render()` to get headings and remarkPluginFrontmatter, then passing them to PostLayout. All posts are passed for prev/next navigation.

The build script now runs `astro build && npx -y pagefind --site dist`. This:
1. First runs astro build to generate the static site
2. Then runs pagefind CLI to index the built HTML into /pagefind/ directory
3. The `-y` flag auto-accepts any prompts
4. Pagefind outputs to `dist/pagefind/` which is already in .gitignore typically

Note: Search will only work after a full build, not in dev mode. The Search component handles this gracefully (Pagefind UI just won't render if files don't exist).
